\section{Schemas}

\begin{schema}{FileMode}
  user: FilePermission \\
  group: FilePermission \\
  owner: FilePermission
\end{schema}

\begin{schema}{Object}
  mode: FileMode \\
  type: FileType \\
	digest: Digest \\
	filename: Name
\end{schema}

\begin{zed}
  Tree ::= blob \ldata Object \rdata | branch \ldata Object \cross \finset Tree \rdata
\end{zed}

\begin{schema}{Commit}
	digest: Digest \\
	author: Author \\
	message: Message \\
	commitTree: Tree
\end{schema}

\begin{axdef}
  hash: Object \surj Digest
\end{axdef}
